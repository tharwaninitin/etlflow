113   def createBigQuerySession(gp: Option[GlobalProperties]): BigQuery = {
114     gp match {
115       case Some(ss) =>
116         ic_logger.info(s"Job is running on env: ${ss.running_environment}")
117         if (ss.running_environment == "gcp") BigQueryOptions.getDefaultInstance.getService
118         else if (ss.running_environment == "aws" || ss.running_environment == "local")  {
119           val credentials: GoogleCredentials  = ServiceAccountCredentials.fromStream(new FileInputStream(ss.gcp_project_key_name))
120           BigQueryOptions.newBuilder().setCredentials(credentials).build().getService
121         }
122         else {
123           throw new Exception("Exception occurred! Please provide correct value of property running_environment in loaddata.properties. Expected values are gcp or aws or local")
124         }
125       case None => BigQueryOptions.getDefaultInstance.getService
